    <script>
        function init() {
            const g = localStorage.getItem('IND_GEMINI_KEY');
            const s = localStorage.getItem('IND_SERPER_KEY');
            const o = localStorage.getItem('IND_OPENROUTER_KEY');
            
            if(g) document.getElementById('key-gemini').value = g;
            if(o) document.getElementById('key-openrouter').value = o;
            if(s) document.getElementById('key-serper').value = s;

            if (!g || !s) {
                document.getElementById('setup-warning').classList.remove('hidden');
            } else {
                document.getElementById('setup-warning').classList.add('hidden');
            }
        }
        window.onload = init;

        function saveKeys() {
            localStorage.setItem('IND_GEMINI_KEY', document.getElementById('key-gemini').value.trim());
            localStorage.setItem('IND_OPENROUTER_KEY', document.getElementById('key-openrouter').value.trim());
            localStorage.setItem('IND_SERPER_KEY', document.getElementById('key-serper').value.trim());
            alert('Keys stored securely on your browser instance.');
            init();
            switchTab('analysis');
        }

        function clearKeys() {
            if(confirm('Wipe local device key cache?')) {
                localStorage.removeItem('IND_GEMINI_KEY');
                localStorage.removeItem('IND_OPENROUTER_KEY');
                localStorage.removeItem('IND_SERPER_KEY');
                document.getElementById('key-gemini').value = "";
                document.getElementById('key-openrouter').value = "";
                document.getElementById('key-serper').value = "";
                init();
            }
        }

        function switchTab(tab) {
            ['analysis', 'penny', 'keys'].forEach(t => {
                const btn = document.getElementById(`tab-${t}-btn`);
                const sec = document.getElementById(`tab-${t}`);
                if (t === tab) {
                    btn.className = `px-4 py-2 text-xs font-bold rounded-lg transition ${t === 'penny' ? 'bg-amber-500 text-slate-950' : 'bg-cyan-500 text-slate-950'}`;
                    sec.classList.remove('hidden');
                } else {
                    btn.className = "px-4 py-2 text-xs font-bold rounded-lg transition text-slate-400 hover:text-white";
                    sec.classList.add('hidden');
                }
            });
        }

        function sleep(ms) {
            return new Promise(resolve => setTimeout(resolve, ms));
        }

        async function fetchIndianMarketData(searchQuery) {
            const key = localStorage.getItem('IND_SERPER_KEY');
            if(!key) return "Error: Missing Serper API Key.";
            
            async function makeRequest(query) {
                try {
                    const res = await fetch("https://google.serper.dev/search", {
                        method: "POST",
                        headers: { "X-API-KEY": key, "Content-Type": "application/json" },
                        body: JSON.stringify({ q: query, gl: "in", hl: "en" })
                    });
                    const data = await res.json();
                    if (data.organic && data.organic.length > 0) {
                        return data.organic.slice(0, 5).map(item => `Title: ${item.title}\nInfo: ${item.snippet}`).join("\n\n");
                    }
                } catch (e) { console.error(e); }
                return null;
            }

            let results = await makeRequest(searchQuery);
            if (!results) {
                results = await makeRequest("top breakout stocks nse under 50 rs moneycontrol");
            }
            if (!results) {
                results = await makeRequest("nse top gainers under 50");
            }
            return results || "Error: No live web context could be scraped from the market engines.";
        }

        async function fetchGemini(prompt, key) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${key}`;
            const res = await fetch(url, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
            });
            const data = await res.json();
            if (data.candidates?.[0]?.content?.parts?.[0]?.text) return data.candidates[0].content.parts[0].text;
            throw new Error("Gemini network error");
        }

        async function fetchOpenRouter(model, prompt, key) {
            const res = await fetch("https://openrouter.ai/api/v1/chat/completions", {
                method: "POST",
                headers: { 
                    "Authorization": `Bearer ${key}`, 
                    "Content-Type": "application/json",
                    "HTTP-Referer": window.location.href,
                    "X-Title": "AI Stock Companion"
                },
                body: JSON.stringify({ model: model, messages: [{ role: "user", content: prompt }] })
            });
            const data = await res.json();
            if (data.choices?.[0]?.message?.content) return data.choices[0].message.content;
            throw new Error(data.error?.message || "Model connection error");
        }

        function updateModelGrid(element, modelName, content) {
            let cleanHtml = parseSimpleMarkdown(content);
            element.innerHTML += `
                <div class="bg-slate-900 p-5 rounded-xl border border-slate-800 shadow-md">
                    <div class="text-xs text-cyan-400 font-bold mb-2 tracking-wide border-b border-slate-800 pb-1">${modelName}</div>
                    <div class="text-xs text-slate-300 leading-relaxed">${cleanHtml}</div>
                </div>
            `;
        }

        async function runMultiModelAnalysis() {
            const ticker = document.getElementById('ticker-input').value.trim().toUpperCase();
            if(!ticker) return alert("Enter an Indian Ticker (e.g. TATAMOTORS)");

            const loader = document.getElementById('analysis-loader');
            const outputBox = document.getElementById('analysis-output-box');
            const consensusTarget = document.getElementById('consensus-verdict-target');
            const grid = document.getElementById('model-responses-grid');

            loader.classList.remove('hidden');
            outputBox.classList.add('hidden');
            grid.innerHTML = "";

            const liveContext = await fetchIndianMarketData(`${ticker} share price target news analysis nse`);
            
            const prompt = `
            You are an Indian stock market advisor. Review this live web data for ticker ${ticker}:
            "${liveContext}"

            Provide a clear verdict using Indian market terminology. You must include this exact string layout in your text:
            VERDICT: [BUY, SELL, or STAY]
            REASON: [One short sentence explaining why in simple English]
            `;

            const gKey = localStorage.getItem('IND_GEMINI_KEY');
            const oKey = localStorage.getItem('IND_OPENROUTER_KEY');

            let buys = 0, sells = 0, stays = 0;
            
            if(gKey) {
                try {
                    const res = await fetchGemini(prompt, gKey);
                    updateModelGrid(grid, "Gemini 2.5 Flash", res);
                    if (res.toUpperCase().includes("VERDICT: BUY")) buys++;
                    else if (res.toUpperCase().includes("VERDICT: SELL")) sells++;
                    else if (res.toUpperCase().includes("VERDICT: STAY")) stays++;
                } catch(e) {
                    updateModelGrid(grid, "Gemini 2.5 Flash", "❌ API Error. Cascading downwards...");
                }
            }

            const orModels = [
                { id: "deepseek/deepseek-r1:free", name: "DeepSeek R1 (Free)" },
                { id: "meta-llama/llama-3.3-70b-instruct:free", name: "Llama 3.3 70B (Free)" },
                { id: "google/gemma-3-27b-it:free", name: "Gemma 3 27B (Free)" }
            ];

            if(oKey) {
                for(let model of orModels) {
                    try {
                        await sleep(800);
                        const res = await fetchOpenRouter(model.id, prompt, oKey);
                        updateModelGrid(grid, model.name, res);
                        if (res.toUpperCase().includes("VERDICT: BUY")) buys++;
                        else if (res.toUpperCase().includes("VERDICT: SELL")) sells++;
                        else if (res.toUpperCase().includes("VERDICT: STAY")) stays++;
                    } catch(e) {
                        updateModelGrid(grid, model.name, `❌ Skipped/Busy: Free tier limit hit.`);
                    }
                }
            }

            let finalDecision = "STAY / HOLD";
            let colorClass = "bg-amber-500/10 text-amber-400 border-amber-500/30";
            if (buys > sells && buys >= stays) { finalDecision = "BUY / ACCUMULATE"; colorClass = "bg-emerald-500/10 text-emerald-400 border-emerald-500/30"; }
            else if (sells > buys && sells >= stays) { finalDecision = "SELL / LIQUIDATE"; colorClass = "bg-rose-500/10 text-rose-400 border-rose-500/30"; }

            consensusTarget.innerHTML = `
                <div class="flex flex-col sm:flex-row sm:items-center justify-between gap-4 mt-2">
                    <div>
                        <div class="text-3xl font-black tracking-tight text-white">${ticker}</div>
                        <div class="text-xs text-slate-400 mt-1">Waterfall Matrix: ${buys} Buy, ${sells} Sell, ${stays} Hold</div>
                    </div>
                    <div class="px-6 py-3 rounded-xl border text-center font-black tracking-wider text-sm ${colorClass}">
                        ${finalDecision}
                    </div>
                </div>
            `;

            loader.classList.add('hidden');
            outputBox.classList.remove('hidden');
        }

        async function runMultiModelPennyScanner() {
            const loader = document.getElementById('penny-loader');
            const outputBox = document.getElementById('penny-output-box');
            const consensusTarget = document.getElementById('penny-consensus-target');
            const grid = document.getElementById('penny-responses-grid');

            loader.classList.remove('hidden');
            outputBox.classList.add('hidden');
            grid.innerHTML = "";

            const liveContext = await fetchIndianMarketData("breakout stocks under 50 rs nse high volume potential profit");

            const prompt = `
            Review this live web data tracking low price high volume breakouts under ₹50:
            "${liveContext}"

            Extract ONE precise Indian ticker symbol poised to deliver the maximum percentage profit return within a strict 1-2 week window.
            Format the response exactly as follows:
            TICKER: [SYMBOL]
            PROFIT TARGET: [Expected % Profit in 1-2 weeks]
            CATALYST: [One short reason sentence explaining the volume spike]
            `;

            const gKey = localStorage.getItem('IND_GEMINI_KEY');
            const oKey = localStorage.getItem('IND_OPENROUTER_KEY');

            let targetStock = "N/A";

            if(gKey) {
                try {
                    const res = await fetchGemini(prompt, gKey);
                    updateModelGrid(grid, "Gemini 2.5 Flash", res);
                    if(res.includes("TICKER:")) {
                        targetStock = res.split("TICKER:")[1]?.split("\n")[0]?.replace(/[^a-zA-Z0-9]/g, "").trim() || "N/A";
                    }
                } catch(e) {
                    updateModelGrid(grid, "Gemini 2.5 Flash", "❌ Pipeline error. Falling forward...");
                }
            }

            if(oKey) {
                const orModels = [
                    { id: "deepseek/deepseek-r1:free", name: "DeepSeek R1 (Free)" },
                    { id: "meta-llama/llama-3.3-70b-instruct:free", name: "Llama 3.3 70B (Free)" }
                ];

                for(let model of orModels) {
                    try {
                        await sleep(800);
                        const res = await fetchOpenRouter(model.id, prompt, oKey);
                        updateModelGrid(grid, model.name, res);
                        if(targetStock === "N/A" && res.includes("TICKER:")) {
                            targetStock = res.split("TICKER:")[1]?.split("\n")[0]?.replace(/[^a-zA-Z0-9]/g, "").trim() || "N/A";
                        }
                    } catch(e) {
                        updateModelGrid(grid, model.name, `❌ Skipped: Model pipeline overloaded.`);
                    }
                }
            }

            if (targetStock === "N/A" || targetStock === "") targetStock = "Scan Complete";

            consensusTarget.innerHTML = `
                <div class="flex flex-col sm:flex-row justify-between items-start sm:items-center gap-4">
                    <div>
                        <div class="text-3xl font-black tracking-tight text-amber-400">${targetStock}</div>
                        <p class="text-xs text-slate-400 mt-1">Highest volatility target extracted via waterfall parameters.</p>
                    </div>
                    <div class="bg-amber-500/10 text-amber-400 border border-amber-500/30 px-4 py-2 rounded-xl text-xs font-bold">
                        ⏱️ Horizon: 1-2 Weeks
                    </div>
                </div>
            `;

            loader.classList.add('hidden');
            outputBox.classList.remove('hidden');
        }

        function parseSimpleMarkdown(text) {
            if (!text) return "";
            return text
                .replace(/^# (.*)$/gim, '
$1
')
                .replace(/VERDICT:\s*BUY/gim, '🟢 BUY')
                .replace(/VERDICT:\s*SELL/gim, '🔴 SELL')
                .replace(/VERDICT:\s*STAY/gim, '🟡 HOLD / STAY')
                .replace(/TICKER:\s*(.*)$/gim, 'Ticker Symbol: $1')
                .replace(/\*\*(.*?)\*\*/g, '$1')
                .replace(/^\* (.*)$/gim, '
  • $1
    • ')
                .replace(/\n/g, '
    ');
        }
